1function [k,v] = myconvhulln(x)
 2%CONVHULLN N-D Convex hull.
 3%   K = CONVHULLN(X) returns the indices K of the points in X that 
 4%   comprise the facets of the convex hull of X. 
 5%   X is an m-by-n array representing m points in n-D space. 
 6%   If the convex hull has p facets then K is p-by-n. 
 7%
 8%   CONVHULLN uses Qhull.
 9%
10%   K = CONVHULLN(X,OPTIONS) specifies a cell array of strings OPTIONS to be
11%   used as options in Qhull. The default options are:
12%                                 {'Qt'} for 2D, 3D and 4D input,
13%                                 {'Qt','Qx'} for 5D and higher. 
14%   If OPTIONS is [], the default options will be used.
15%   If OPTIONS is {''}, no options will be used, not even the default.
16%   For more information on Qhull and its options, see http://www.qhull.org.
17%
18%   [K,V] = CONVHULLN(...) also returns the volume of the convex hull
19%   in V. 
20%
21%   Example:
22%      X = [0 0; 0 1e-10; 0 0; 1 1];
23%      K = convhulln(X)
24%   gives a warning that is suppresed by the additional option 'Pp':
25%      K = convhulln(X,{'Qt','Pp'})
26%
27%   See also delaunayTriangulation, triangulation, CONVHULL, QHULL, DELAUNAYN,
28%            VORONOIN, TSEARCHN, DSEARCHN.
29
30%   Copyright 1984-2013 The MathWorks, Inc. 
31
32if nargin < 1
33    error(message('MATLAB:convhulln:NotEnoughInputs'));
34end
35
36cgprechecks(x, nargin);
37
38n = size(x,2);
39if n <= 1
40  error(message('MATLAB:convhulln:XLowColNum'));
41end
42
43[x, dupesfound, idxmap] = mergeDuplicatePoints(x);
44n = size(x,2);
45
46%default options
47if n >= 5
48    opt = 'Qt Qx';
49else 
50    opt = 'Qt';
51end
52
53[k,vv] = qhullmx(x', opt);
54
55if nargout > 1
56    v = vv;
57end
58
59if n == 2
60    % Sort the vertices in counterclockwise order
61    k = unique(k(:));
62    k = k(:);
63    xc = x(k,1); yc = x(k,2);
64    xm = mean(xc); ym = mean(yc);
65    phi = angle((xc-xm) + (yc-ym)*sqrt(-1));
66    [~,ind] = unique(phi);
67    k = k(ind);
68    num_v = length(k);
69    k = [k(1:(num_v-1)) k(2:num_v); k(num_v) k(1)];    
70end
71    
72% Rewire the vertex indices if points were merged
73if (dupesfound)
74    k = idxmap(k);
75end
