1function [Xout, dupesfound, idxmap] = mergeDuplicatePoints(X)
 2%MergeDuplicatePoints Merge out points that have coincident location.
 3
 4%   Copyright 2009 The MathWorks, Inc.
 5
 6dupesfound = false;
 7numinitpoints = size(X, 1);
 8[~,idxmap] = unique(X,'first','rows');
 9numuniquepoints = length(idxmap);
10if (numinitpoints > numuniquepoints)
11    % Undo the sort to preserve the ordering of points
12    idxmap2 = sort(idxmap)';
13    Xout = X(idxmap2,:);    
14    dupesfound = true;
15else
16    Xout = X;
17end
