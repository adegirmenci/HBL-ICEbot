1function in = myinhull(testpts,xyz)
  2% inhull: tests if a set of points are inside a convex hull
  3% usage: in = inhull(testpts,xyz)
  4% usage: in = inhull(testpts,xyz,tess)
  5% usage: in = inhull(testpts,xyz,tess,tol)
  6%
  7% arguments: (input)
  8%  testpts - nxp array to test, n data points, in p dimensions
  9%       If you have many points to test, it is most efficient to
 10%       call this function once with the entire set.
 11%
 12%  xyz - mxp array of vertices of the convex hull, as used by
 13%       convhulln.
 14%
 15%  tess - tessellation (or triangulation) generated by convhulln
 16%       If tess is left empty or not supplied, then it will be
 17%       generated.
 18%
 19%  tol - (OPTIONAL) tolerance on the tests for inclusion in the
 20%       convex hull. You can think of tol as the distance a point
 21%       may possibly lie outside the hull, and still be perceived
 22%       as on the surface of the hull. Because of numerical slop
 23%       nothing can ever be done exactly here. I might guess a
 24%       semi-intelligent value of tol to be
 25%
 26%         tol = 1.e-13*mean(abs(xyz(:)))
 27%
 28%       In higher dimensions, the numerical issues of floating
 29%       point arithmetic will probably suggest a larger value
 30%       of tol.
 31%
 32%       DEFAULT: tol = 0
 33%
 34% arguments: (output)
 35%  in  - nx1 logical vector
 36%        in(i) == 1 --> the i'th point was inside the convex hull.
 37%  
 38% Example usage: The first point should be inside, the second out
 39%
 40%  xy = randn(20,2);
 41%  tess = convhulln(xy);
 42%  testpoints = [ 0 0; 10 10];
 43%  in = inhull(testpoints,xy,tess)
 44%
 45% in = 
 46%      1
 47%      0
 48%
 49% A non-zero count of the number of degenerate simplexes in the hull
 50% will generate a warning (in 4 or more dimensions.) This warning
 51% may be disabled off with the command:
 52%
 53%   warning('off','inhull:degeneracy')
 54%
 55% See also: convhull, convhulln, delaunay, delaunayn, tsearch, tsearchn
 56%
 57% Author: John D'Errico
 58% e-mail: woodchips@rochester.rr.com
 59% Release: 3.0
 60% Release date: 10/26/06
 61
 62% get array sizes
 63% m points, p dimensions
 64coder.allowpcode('plain');
 65
 66p = size(xyz,2);
 67[n,c] = size(testpts);
 68if p ~= c
 69  error 'testpts and xyz must have the same number of columns'
 70end
 71if p < 2
 72  error 'Points must lie in at least a 2-d space.'
 73end
 74
 75tess = myconvhulln(xyz);
 76
 77[nt,c] = size(tess);
 78
 79if c ~= p
 80    disp(c)
 81  error 'tess array is incompatible with a dimension p space'
 82end
 83
 84tol = 0;
 85
 86% build normal vectors
 87switch p
 88  case 2
 89    % really simple for 2-d
 90    nrmls = (xyz(tess(:,1),:) - xyz(tess(:,2),:)) * [0 1;-1 0];
 91    
 92    % Any degenerate edges?
 93    del = sqrt(sum(nrmls.^2,2));
 94    degenflag = (del<(max(del)*10*eps));
 95    if sum(degenflag)>0
 96        printf('inhull:degeneracy %d degenerate edges identified in the convex hull', sum(degenflag))
 97      
 98      % we need to delete those degenerate normal vectors
 99      nrmls(degenflag,:) = [];
100      nt = size(nrmls,1);
101    end
102  case 3
103    % use vectorized cross product for 3-d
104    ab = xyz(tess(:,1),:) - xyz(tess(:,2),:);
105    ac = xyz(tess(:,1),:) - xyz(tess(:,3),:);
106    nrmls = cross(ab,ac,2);
107    degenflag = false(nt,1);
108  otherwise
109    % slightly more work in higher dimensions, 
110    nrmls = zeros(nt,p);
111    degenflag = false(nt,1);
112    for i = 1:nt
113      % just in case of a degeneracy
114      % Note that bsxfun COULD be used in this line, but I have chosen to
115      % not do so to maintain compatibility. This code is still used by
116      % users of older releases.
117      %  nullsp = null(bsxfun(@minus,xyz(tess(i,2:end),:),xyz(tess(i,1),:)))';
118      nullsp = null(xyz(tess(i,2:end),:) - repmat(xyz(tess(i,1),:),p-1,1))';
119      if size(nullsp,1)>1
120        degenflag(i) = true;
121        nrmls(i,:) = NaN;
122      else
123        nrmls(i,:) = nullsp;
124      end
125    end
126    if sum(degenflag)>0
127        printf('inhull:degeneracy %d degenerate simplexes identified in the convex hull', sum(degenflag))
128      
129      % we need to delete those degenerate normal vectors
130      nrmls(degenflag,:) = [];
131      nt = size(nrmls,1);
132    end
133end
134
135% scale normal vectors to unit length
136nrmllen = sqrt(sum(nrmls.^2,2));
137% again, bsxfun COULD be employed here...
138%  nrmls = bsxfun(@times,nrmls,1./nrmllen);
139nrmls = nrmls.*repmat(1./nrmllen,1,p);
140
141% center point in the hull
142center = mean(xyz,1);
143
144% any point in the plane of each simplex in the convex hull
145a = xyz(tess(~degenflag,1),:);
146
147% ensure the normals are pointing inwards
148% this line too could employ bsxfun...
149%  dp = sum(bsxfun(@minus,center,a).*nrmls,2);
150dp = sum((repmat(center,nt,1) - a).*nrmls,2);
151k = dp<0;
152nrmls(k,:) = -nrmls(k,:);
153
154% We want to test if:  dot((x - a),N) >= 0
155% If so for all faces of the hull, then x is inside
156% the hull. Change this to dot(x,N) >= dot(a,N)
157aN = sum(nrmls.*a,2);
158
159% test, be careful in case there are many points
160in = false(n,1);
161
162% if n is too large, we need to worry about the
163% dot product grabbing huge chunks of memory.
164memblock = 1e6;
165blocks = max(1,floor(n/(memblock/nt)));
166aNr = repmat(aN,1,length(1:blocks:n));
167for i = 1:blocks
168   j = i:blocks:n;
169   if size(aNr,2) ~= length(j),
170      aNr = repmat(aN,1,length(j));
171   end
172   in(j) = all((nrmls*testpts(j,:)' - aNr) >= -tol,1)';
173end
